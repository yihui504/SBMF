### 3.2 创建实战测试脚本

创建 `tests/integration/test_e2e实战.py`:

```python
"""
端到端实战测试 - 使用模拟 SeekDB Adapter
验证完整框架流程
"""
import pytest
from contract import load_contract
from profiles import SeekDBProfilePlugin, register_profile
from core import ExecutionPipeline, OracleReporter, RuleEngine, PreconditionGate
from core.models import SemanticCase, SlotScope
from oracle import RangeConstraintOracle, EnumConstraintOracle, RelationalConstraintOracle
from adapters.seekdb import SeekDBAdapter

def test_end_to_end_normal_flow():
    """测试正常流程：合法操作成功执行"""
    # 1. 加载 Contract
    contract = load_contract("tests/fixtures/integration/seekdb_contract.yaml")

    # 2. 创建组件
    adapter = SeekDBAdapter()  # 模拟 adapter
    profile = SeekDBProfilePlugin(enable_logging=False)

    rule_engine = RuleEngine(contract)
    precondition_gate = PreconditionGate(rule_engine)

    # 3. 配置 Oracle 检查器
    oracles = [
        RangeConstraintOracle("dimension", min_value=1, max_value=32768),
        EnumConstraintOracle("metric_type", allowed_values=["L2", "IP", "COSINE"]),
        RelationalConstraintOracle("search_range", ComparisonOperator.GE, "top_k"),
    ]

    pipeline = ExecutionPipeline(
        rule_engine=rule_engine,
        precondition_gate=precondition_gate,
        oracles=oracles
    )

    # 4. 创建测试用例
    test_case = SemanticCase(
        test_id="实战_001",
        operation="search",
        slot_values={
            "dimension": 512,
            "metric_type": "L2",
            "top_k": 10,
            "search_range": 100,
        },
        raw_parameters={},
        is_legal=True,
        scope=SlotScope.COLLECTION
    )

    # 5. 执行测试
    result = pipeline.execute_test_case(
        test_case=test_case,
        contract=contract,
        adapter=adapter,
        profile=profile
    )

    # 6. 验证结果
    assert result.status.value == "SUCCESS"
    assert len(result.oracle_results) == 3
    assert all(r.passed for r in result.oracle_results)

    print(f"✅ 测试通过: {test_case.test_id}")
    print(f"   Oracle 结果: {len(result.oracle_results)} 个检查全部通过")

def test_end_to_end_semantic_violation():
    """测试语义违背检测"""
    contract = load_contract("tests/fixtures/integration/seekdb_contract.yaml")
    adapter = SeekDBAdapter()
    profile = SeekDBProfilePlugin(enable_logging=False)

    rule_engine = RuleEngine(contract)
    precondition_gate = PreconditionGate(rule_engine)

    oracles = [
        RangeConstraintOracle("dimension", min_value=1, max_value=32768),
    ]

    pipeline = ExecutionPipeline(
        rule_engine=rule_engine,
        precondition_gate=precondition_gate,
        oracles=oracles
    )

    # 创建违反约束的测试用例
    test_case = SemanticCase(
        test_id="实战_002_维度超限",
        operation="search",
        slot_values={
            "dimension": 99999,  # 超过最大值 32768
            "metric_type": "L2",
            "top_k": 10,
        },
        raw_parameters={},
        is_legal=True,
        scope=SlotScope.COLLECTION
    )

    result = pipeline.execute_test_case(
        test_case=test_case,
        contract=contract,
        adapter=adapter,
        profile=profile
    )

    # Profile 应该跳过
    assert result.status.value == "PRECONDITION_FAILED"
    assert "dimension" in result.gate_result.reason.lower() or "99999" in result.gate_result.reason

    print(f"✅ 正确检测到约束违反: {test_case.test_id}")
    print(f"   跳过原因: {result.gate_result.reason}")
```
